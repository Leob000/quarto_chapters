\tikzmath{
  function logistic(\x) {
    if \x > 0 then {
      return 1 / (1 + exp(-\x));
    } else {
      return 1 * exp(\x) / (1 + exp(\x));
    };
  };
}

  \pgfmathsetmacro{\x}{#1}
  \draw [color=#2, line width=1]
      (\x, -2) .. controls (\x - 0.75, -1) and (\x + 0.5, -0.25)

}
