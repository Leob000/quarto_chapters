\tikzmath{
  function logistic(\x) {
    if \x > 0 then {
      return 1 / (1 + exp(-\x));
    } else {
      return 1 * exp(\x) / (1 + exp(\x));
    };
  };
}

\newcommand{\fiber}[2]{
  \pgfmathsetmacro{\x}{#1}
  \draw [color=#2, line width=1]
      (\x, -2) .. controls (\x - 0.75, -1) and (\x + 0.5, -0.25) 
   .. (\x + 0.5, 0.25) .. controls (\x + 0.5, 0.75) and (\x - 0.5, 1.25) .. (\x + 0.25, 2)
}

\newcommand{\randpoints}[3]{
