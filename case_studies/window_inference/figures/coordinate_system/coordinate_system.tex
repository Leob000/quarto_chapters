  \pgfmathsetmacro{\D}{4}
  \pgfmathsetmacro{\R}{0.075}

  \draw[black, line width=1] (-\D, -\D) rectangle (+\D, +\D);
  
  \fill[dark] (-\D, -\D) circle (\R)
  node[below left] { $(-L, -L)$ };
  
  \fill[dark] (-\D, +\D) circle (\R)
  node[above left] { $(-L, +L)$ };
  
  \fill[dark] (+\D, -\D) circle (\R)
  node[below right] { $(+L, -L)$ };
  
  \fill[dark] (+\D, +\D) circle (\R)
  node[above right] { $(+L, +L)$ };
  
  \fill[dark] (0, 0) circle (\R)
  node[below right] { $(0, 0)$ };
  
  % Window
  \pgfmathsetmacro{\l}{-0.25}
  \pgfmathsetmacro{\r}{2.75}
  \pgfmathsetmacro{\b}{-1}
  \pgfmathsetmacro{\t}{2}
  
  \draw[black, line width=1] (\l, \b) rectangle (\r,\t);
    
  \fill[dark] (\l, \b) circle (\R)
  node[below left] { $(l, b)$ };
  
  \fill[dark] (\r, \b) circle (\R)
  node[below right] { $(r, b)$ };
  
  \fill[dark] (\l, \t) circle (\R)
  node[above left] { $(l, t)$ };
  
  \fill[dark] (\r, \t) circle (\R)
  node[above right] { $(r, t)$ };
  
  \draw[light, line width=1, decorate,decoration={brace,amplitude=3pt}] 
    (\l, \t + 0.1) -- (\r, \t + 0.1);
  \node[light] at ( {(\r + \l) / 2}, \t + 0.5) { $r - l$ };
      
  \draw[light, line width=1, decorate,decoration={brace,amplitude=3pt}] 
    (\l - 0.1, \b) -- (\l - 0.1, \t);
  \node[light] at (\l - 0.75, {(\t + \b) / 2}) { $t - b$ };
