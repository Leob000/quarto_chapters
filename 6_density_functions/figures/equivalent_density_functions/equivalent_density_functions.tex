\tikzmath{
  function normal(\x, \m, \s) {
    return exp(-0.5 * (\x - \m) * (\x - \m) / (\s * \s) ) / (2.506628274631001 * \s);
  };
  function density(\x) {
    return 0.6 * normal(\x, 0, 0.8) + 0.2 * normal(\x, 1, 0.35) + 0.2 * normal(\x, -1.5, 0.35);
  };
}
