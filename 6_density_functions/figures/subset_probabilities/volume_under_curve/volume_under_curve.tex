\tikzmath{
  function normal(\x, \m, \s) {
    return exp(-0.5 * (\x - \m) * (\x - \m) / (\s * \s) ) / (2.506628274631001 * \s);
  };
  function indicator(\x, \y) {
    if sqrt( (\x - 1) * (\x - 1) + (\y + 1) * (\y + 1) ) < 0.25 then {
      return 1;
    } else {
      return 0;
    };
  };
}
