\tikzmath{
  function f(\x) {
    return sin(100 * \x) + 0.25 * \x;
  };
  function fp(\x) {
    \y = sin(100 * \x) + 0.25 * \x;
    if \y > 0 then {
      return \y;
    } else {
      return 0;
    };
  };
  function fm(\x) {
    \y = sin(100 * \x) + 0.25 * \x;
    if \y < 0 then {
      return \y;
    } else {
      return 0;
    };
  };
}
