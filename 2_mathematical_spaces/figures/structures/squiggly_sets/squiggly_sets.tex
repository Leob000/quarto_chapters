% #1: R
% #2 - #9: Radial Perturbations
\newcommand{\points}[9]{
  ({\xo + #2 * #1 * (-1)},     {\yo + #2 * #1 * (0)})
  ({\xo + #3 * #1 * (-0.7071)}, {\yo + #3 * #1 * (+0.7071)})
  ({\xo + #4 * #1 * (0)},      {\yo + #4 * #1 * (+1)})
  ({\xo + #5 * #1 * (+0.7071)}, {\yo + #5 * #1 * (+0.7071)})
  ({\xo + #6 * #1 * (+1)},     {\yo + #6 * #1 * (0)})
  ({\xo + #7 * #1 * (+0.7071)}, {\yo + #7 * #1 * (-0.7071)})
  ({\xo + #8 * #1 * (0)},      {\yo + #8 * #1 * (-1)})
  ({\xo + #9 * #1 * (-0.7071)}, {\yo + #9 * #1 * (-0.7071)})
}

% #1: x0
% #2: y0
% #3: R
\newcommand{\randpoints}[3]{
  ({#1 + (1 + 0.2 * rand) * #3 * (-1)},      {#2 + (1 + 0.2 * rand) * #3 * (0)})
  ({#1 + (1 + 0.2 * rand) * #3 * (-0.7071)}, {#2 + (1 + 0.2 * rand) * #3 * (+0.7071)})
  ({#1 + (1 + 0.2 * rand) * #3 * (0)},       {#2 + (1 + 0.2 * rand) * #3 * (+1)})
  ({#1 + (1 + 0.2 * rand) * #3 * (+0.7071)}, {#2 + (1 + 0.2 * rand) * #3 * (+0.7071)})
  ({#1 + (1 + 0.2 * rand) * #3 * (+1)},      {#2 + (1 + 0.2 * rand) * #3 * (0)})
  ({#1 + (1 + 0.2 * rand) * #3 * (+0.7071)}, {#2 + (1 + 0.2 * rand) * #3 * (-0.7071)})
  ({#1 + (1 + 0.2 * rand) * #3 * (0)},       {#2 + (1 + 0.2 * rand) * #3 * (-1)})
  ({#1 + (1 + 0.2 * rand) * #3 * (-0.7071)}, {#2 + (1 + 0.2 * rand) * #3 * (-0.7071)})
}

\pgfmathsetseed{12}

  \draw[white] (-3.5, -3.5) rectangle (3.5, 3.5);

  \foreach [count=\n] \r in {0.5, 1.25, ..., 3} {
    \pgfmathsetmacro{\prop}{100 * \n / 2.5}
    \colorlet{custom}{dark!\prop!light}; 
    \draw[custom, dashed, line width=1] 
      plot [smooth cycle, tension=0.75] coordinates { \randpoints{0}{0}{\r} } -- cycle;
  }
