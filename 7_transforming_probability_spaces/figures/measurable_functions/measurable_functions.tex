\tikzmath{
  function f(\x) {
    return sin(100 * \x) + 0.25 * \x;
  };
}

\begin{document}

\begin{tikzpicture}[scale=1.0]

  \begin{scope}[shift={(0, 0)}]
    \draw[white] (-3.75, -2.25) rectangle (3.75, 3);
    
    \draw [<->, >=stealth, line width=1.25] (0, -2.015) -- +(0, 4);
    \draw [<->, >=stealth, line width=1.25] (-3.015, 0.00) -- +(6, 0);
    \node at (0, 2.5) { $f(x)$ };
    \node at (3.25, 0) { $x$ };
    
    \pgfmathsetmacro{\prop}{40};
    \colorlet{custom}{dark!\prop!white};
    \draw[domain={-3:-1}, smooth, samples=100, line width=1.25, variable=\x, color=custom] 
      plot ({\x},{+0.75 * f(1.25 * \x) + 1.75}); 
    \draw[domain={-1:3}, smooth, samples=100, line width=1.25, variable=\x, color=custom] 
      plot ({\x},{-0.75 * f(1.25 * \x) - 0.75}); 
    
    \pgfmathsetmacro{\prop}{70};
    \colorlet{custom}{dark!\prop!white};
    \draw[domain={-3:1.515}, smooth, samples=100, line width=1.25, variable=\x, color=custom] 
      plot ({\x},{-f(0.5 * (\x - 1.5)) - 0.5}); 
    \draw[domain={1.485:3}, smooth, samples=100, line width=1.25, variable=\x, color=custom] 
      plot ({\x},{+f(0.5 * (\x - 1.5)) - 0.5}); 
    
    \pgfmathsetmacro{\prop}{100};
    \colorlet{custom}{dark!\prop!white};
    \draw[domain={-3:3}, smooth, samples=100, line width=1.25, variable=\x, color=custom] 
      plot ({\x},{f(\x)});                         
  \end{scope}
  
\end{tikzpicture}
