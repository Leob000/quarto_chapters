\tikzmath{
  function normal(\x) {
    return exp(-0.5 * \x * \x);
  };
  function logistic(\x) {
    if \x > 0 then {
      return 1 / (1 + exp(-\x));
    } else {
      return 1 * exp(\x) / (1 + exp(\x));
    };
  };
}
